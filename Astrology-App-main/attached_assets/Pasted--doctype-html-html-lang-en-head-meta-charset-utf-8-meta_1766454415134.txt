  function reduceGG(n){
    while(true){
      if(n===11 || n===22 || n===33) return n;
      if(n<10) return n;
      n = String(n).split("").reduce((a,d)=>a+Number(d),0);
    }
  }

  function sumDigitsOfString(s){
    return s.split("").reduce((a,ch)=>a + (/\d/.test(ch)?Number(ch):0),0);
  }

  function getLifePath(d){
    const mm = String(d.getMonth()+1).padStart(2,"0");
    const dd = String(d.getDate()).padStart(2,"0");
    const yyyy = String(d.getFullYear());
    const raw = sumDigitsOfString(mm+dd+yyyy);
    return { rawTotal: raw, lifePath: reduceGG(raw), mm, dd, yyyy };
  }

  function getVietnamAnimal(year){
    let idx = (year - 2020) % 12;
    if(idx<0) idx += 12;
    return animals[idx];
  }

  function nextYearsByAnimals(targetAnimals, count, startYear){
    const out = [];
    let y = startYear;
    while(out.length < count){
      const a = getVietnamAnimal(y);
      if(targetAnimals.includes(a)) out.push({year:y, animal:a});
      y += 1;
    }
    return out;
  }

  function placeNumerology(name){
    const clean = (name||"").toUpperCase().replace(/[^A-Z]/g,"");
    let sum=0, vs=0, cs=0;
    for(const ch of clean){
      const v = letterMap[ch] || 0;
      sum += v;
      if(vowels.has(ch)) vs += v; else cs += v;
    }
    return { clean, sum, reduced: reduceGG(sum), vowelSum: vs, consonantSum: cs };
  }

  // Compatibility scoring
  function relationType(a,b){
    if(a===b) return "same";
    const comp = {
      1:[3,5,7], 2:[4,6,8], 3:[1,5,9], 4:[2,8], 5:[1,3,7], 6:[2,9], 7:[1,5], 8:[2,4], 9:[3,6]

      1:[2,6], 2:[1,5], 3:[4,8], 4:[3,9], 5:[2,8], 6:[1,7], 7:[6,8], 8:[3,5,7], 9:[4]
    };
    if((comp[a]||[]).includes(b)) return "complementary";
    if((fric[a]||[]).includes(b)) return "friction";
    return "neutral";
  }

  function scorePair(a,b,maxSame,maxComp,maxNeutral,maxFric){
    const t = relationType(a,b);
    if(t==="same") return maxSame;
    if(t==="complementary") return maxComp;
    if(t==="neutral") return maxNeutral;
    return maxFric;
  }

  function zodiacScore(animalA, animalB){
    if(animalA===animalB) return 28;

    const sec = (enemySecondary[animalA]||[]).includes(animalB);
    if(prim) return 0;
    if(sec) return 10;
    return 20;
  }

      if(md){
        const d = new Date(md+"T00:00:00");
        const energy = reduceGG(sumDigitsOfString(
          String(d.getFullYear()) + String(d.getMonth()+1).padStart(2,"0") + String(d.getDate()).padStart(2,"0")
        ));
        if(energy===rep.numerology.lifePath) dateScore = 30;
        else if(energy===rep.numerology.dayNum) dateScore = 24;
        else if(energy===rep.numerology.monthNum) dateScore = 20;
        else if(energy===rep.numerology.compositeReduced) dateScore = 20;
        else dateScore = 12;
      }

      const total = Math.round(Math.min(100, placeScoreBase + dateScore));
      document.getElementById("compatResult").innerHTML = `
        <div class="big">${total}</div>
        <div class="muted">Place fit score 0–100</div>
        <div class="hr"></div>
        <div class="muted">Place numerology</div>
        <div>${renderPills([
          "Place number "+pn.reduced,
          "Vowels "+pn.vowelSum,
          "Consonants "+pn.consonantSum
        ])}</div>
        <div class="muted">Match logic</div>
        <div>${renderPills([
          "Life path "+rep.numerology.lifePath+" vs place "+pn.reduced+" → "+relationType(rep.numerology.lifePath, pn.reduced),
          "Base "+placeScoreBase+"/70",
          md ? ("Date alignment "+dateScore+"/30") : "No move date"
        ])}</div>
      `;
    };
  }
